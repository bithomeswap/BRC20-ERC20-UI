# 安装go
wget https://go.dev/dl/go1.22.0.linux-amd64.tar.gz
rm -rf /usr/local/go
tar -C /usr/local -xzf go1.22.0.linux-amd64.tar.gz
export PATH=$PATH:/usr/local/go/bin
go version # 输出版本号说明安装成功

#初始化模块【在任务文件夹下】
go mod init

#运行go文件【或者直接点调试按钮也可以】
go run /root/test/BRC20、ERC20、UI/GO/hellow.go#或者直接点调试按钮也可以

#更新GLIBC库
cd
wget http://ftp.gnu.org/gnu/glibc/glibc-2.39.tar.gz
tar xf glibc-2.39.tar.gz 
cd glibc-2.39/ && mkdir build  && cd build 
# 替换这句话也可以cd ~/glibc-2.39/build
# 配置环境
../configure --prefix=/usr --disable-profile --enable-add-ons --with-headers=/usr/include --with-binutils=/usr/bin
# 安装
make
make install
